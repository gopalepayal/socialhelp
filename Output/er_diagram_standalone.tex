\documentclass[border=15pt, tikz]{standalone}
\usetikzlibrary{shapes.geometric, arrows.meta, positioning, calc}

\begin{document}
\begin{tikzpicture}[
    font=\sffamily,
    entity/.style={
        rectangle, draw=black, thick, fill=white, 
        minimum width=3.2cm, minimum height=1.2cm, align=center
    },
    relationship/.style={
        diamond, draw=black, thick, fill=white, 
        aspect=1.5, align=center, inner sep=2pt, font=\footnotesize
    },
    line/.style={draw=black, thick}
]

% 1. Core Center: Donation
\node[entity] (donation) at (0, 0) {Donation\\(Id, Type, Status)};

% 2. Main Actors
\node[entity] (donor) at (-8, 0) {Donor\\(Id, Name, Email)};
\node[entity] (org) at (8, 0) {Organisation\\(Id, Name, Status)};
\node[entity] (admin) at (0, 7) {Admin\\(Id, Name, Role)};

% 3. Peripheral Entities
\node[entity] (message) at (0, -7) {ChatMessage\\(Id, Sender, Text)};
\node[entity] (requirement) at (14, 0) {Requirement\\(Id, Item, Qty)};
\node[entity] (verification) at (8, 7) {Verification\\(Id, DocPath, Date)};

% --- Relationships ---

% Donor -> Donation (Owns)
\node[relationship] (Rel1) at ($(donor)!0.5!(donation)$) {Owns};
\draw[line] (donor) -- node[above] {1} (Rel1) -- node[above] {N} (donation);

% Org -> Donation (Receives)
\node[relationship] (Rel2) at ($(org)!0.5!(donation)$) {Receives};
\draw[line] (org) -- node[above] {1} (Rel2) -- node[above] {N} (donation);

% Donation -> Message (Has)
\node[relationship] (Rel3) at ($(donation)!0.5!(message)$) {Has};
\draw[line] (donation) -- node[right] {1} (Rel3) -- node[right] {N} (message);

% Org -> Requirement (Posts)
\node[relationship] (Rel4) at ($(org)!0.5!(requirement)$) {Posts};
\draw[line] (org) -- node[above] {1} (Rel4) -- node[above] {N} (requirement);

% Admin -> Donation (Moderates)
\node[relationship] (Rel5) at ($(admin)!0.5!(donation)$) {Moderates};
\draw[line] (admin) -- node[right] {1} (Rel5) -- node[right] {N} (donation);

% Org -> Verification (Submits)
\node[relationship] (Rel6) at ($(org)!0.5!(verification)$) {Submits};
\draw[line] (org) |- node[right, pos=0.8] {1} (Rel6) -- node[above] {1} (verification);

% Admin -> Verification (Approves)
\node[relationship] (Rel7) at ($(admin)!0.5!(verification)$) {Approves};
\draw[line] (admin) -- node[above] {1} (Rel7) -- node[above] {1} (verification);

% Admin -> Donor (Manages)
\node[relationship] (Rel8) at (-8, 7) {Manages};
\draw[line] (admin) -- node[above] {1} (Rel8) -- node[left] {N} (donor);

\end{tikzpicture}
\end{document}
